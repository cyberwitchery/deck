\section{the model is the contract}

% --- itemize -----------------------------------------------------------------
\begin{frame}{four steps, every time}
  \begin{itemize}\setlength\itemsep{1.1em}
    \item describe the shape once\\[2pt]
      {\prose{\small\color{cwFaint}vendor-neutral, set in one reviewable place.}}
    \item pull from the systems you already run\\[2pt]
      {\prose{\small\color{cwFaint}netbox, nautobot, infrahub --- read straight through.}}
    \item map, then apply\\[2pt]
      {\prose{\small\color{cwFaint}see the plan before anything is written back.}}
  \end{itemize}
\end{frame}

% --- blocks ------------------------------------------------------------------
\begin{frame}{three kinds of emphasis}
  \begin{block}{block --- the model is the contract}
    describe it once; every source bends to fit the same shape.
  \end{block}
  \begin{alertblock}{alertblock --- nothing changes by surprise}
    the plan is shown in full before a single write goes out.
  \end{alertblock}
  \begin{exampleblock}{exampleblock --- migrate netbox $\rightarrow$ infrahub}
    same engine, different endpoints. set the map, run it.
  \end{exampleblock}
\end{frame}

% --- two columns + map panel -------------------------------------------------
\begin{frame}{one shape, many endpoints}
  \begin{columns}[T]
    \begin{column}{0.44\textwidth}
      \begin{itemize}\setlength\itemsep{0.8em}
        \item inventory is the source of truth
        \item the fabric is the destination
        \item alembic is the flow between
      \end{itemize}
      \vskip10pt\prose{\small\color{cwFaint}the same system is an input one day and an output the next.}
    \end{column}
    \begin{column}{0.48\textwidth}
      \cwpanel{\color{cwBody}\small
        \cwmaprow{in}{netbox}
        {\color{cwRule}\hrule height0.6pt}%
        \cwmaprow{map}{device model}
        {\color{cwRule}\hrule height0.6pt}%
        \cwmaprow{out}{\color{cwInk}infrahub $\rightarrow$ fabric}}
    \end{column}
  \end{columns}
\end{frame}
